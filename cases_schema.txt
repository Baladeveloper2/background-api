CREATE TABLE "cases" (
  "id" varchar(36) NOT NULL,
  "case_ref_no" varchar(50) DEFAULT NULL,
  "customer_id" varchar(36) DEFAULT NULL,
  "candidate_id" varchar(36) DEFAULT NULL,
  "batch_id" varchar(36) DEFAULT NULL,
  "status" varchar(50) DEFAULT NULL,
  "received_date" datetime DEFAULT (now()),
  "completed_date" datetime DEFAULT NULL,
  "tat_days" int DEFAULT NULL,
  "assigned_to" varchar(36) DEFAULT NULL,
  "qa_id" varchar(36) DEFAULT NULL,
  "qc_id" varchar(36) DEFAULT NULL,
  "assigned_at" datetime DEFAULT NULL,
  "assigned_id" int DEFAULT NULL,
  "ai_summary" text,
  "verifier_revoke_count" int DEFAULT '0',
  "qc_revoke_count" int DEFAULT '0',
  "is_in_tat" int DEFAULT '1',
  "file_no" varchar(50) DEFAULT NULL,
  "insufficiency_count" int DEFAULT '0',
  PRIMARY KEY ("id"),
  UNIQUE KEY "ix_cases_case_ref_no" ("case_ref_no"),
  KEY "ix_cases_assigned_to" ("assigned_to"),
  KEY "ix_cases_batch_id" ("batch_id"),
  KEY "ix_cases_candidate_id" ("candidate_id"),
  KEY "ix_cases_completed_date" ("completed_date"),
  KEY "ix_cases_customer_id" ("customer_id"),
  KEY "ix_cases_received_date" ("received_date"),
  KEY "ix_cases_status" ("status"),
  KEY "ix_cases_qa_id" ("qa_id"),
  KEY "ix_cases_qc_id" ("qc_id"),
  KEY "ix_cases_file_no" ("file_no"),
  CONSTRAINT "cases_ibfk_1" FOREIGN KEY ("customer_id") REFERENCES "customers" ("id") ON DELETE CASCADE,
  CONSTRAINT "cases_ibfk_2" FOREIGN KEY ("candidate_id") REFERENCES "candidates" ("id") ON DELETE CASCADE
)
